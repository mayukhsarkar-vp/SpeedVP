function hav(a,b,c,d){
  const R=6371000,dL=(c-a)*Math.PI/180,dO=(d-b)*Math.PI/180;
  const x=Math.sin(dL/2)**2+Math.cos(a*Math.PI/180)*Math.cos(c*Math.PI/180)*Math.sin(dO/2)**2;
  return R*2*Math.atan2(Math.sqrt(x),Math.sqrt(1-x));
}

  const maxV=UNITS[S.unit].max;const pts=S.speedHistory;const gold=gv('--gold')||'#c9a84c';
  // grid
  for(let i=1;i<4;i++){const y=h-h*(i/4);c.strokeStyle='rgba(255,255,255,.04)';c.lineWidth=1;c.beginPath();c.moveTo(0,y);c.lineTo(w,y);c.stroke();}
  // limit line
  if(S.limit>0){
    const ly=h-h*(S.limit/maxV);
    c.strokeStyle='rgba(255,150,0,.3)';c.lineWidth=1;c.setLineDash([4,4]);
    c.beginPath();c.moveTo(0,ly);c.lineTo(w,ly);c.stroke();c.setLineDash([]);
  }
  // fill
  const gr=c.createLinearGradient(0,0,0,h);gr.addColorStop(0,`${gold}55`);gr.addColorStop(1,`${gold}00`);
  c.beginPath();c.moveTo(0,h);
  pts.forEach((v,i)=>{const x=i*(w/(pts.length-1)),y=h-h*(v/maxV);i===0?c.moveTo(x,y):c.lineTo(x,y);});
  c.lineTo(w,h);c.closePath();c.fillStyle=gr;c.fill();
  // line
  c.beginPath();
  pts.forEach((v,i)=>{const x=i*(w/(pts.length-1)),y=h-h*(v/maxV);i===0?c.moveTo(x,y):c.lineTo(x,y);});
  c.strokeStyle=gold;c.lineWidth=1.5;c.stroke();
}
